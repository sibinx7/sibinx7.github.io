TypeScript

TypeScript is an open-source programming language developed by Microsoft. It is a superset of JavaScript that adds static typing, interfaces, generics, and other features to make applications more maintainable and scalable.

Advantages

  • Static type checking
  • Better IntelliSense and IDE support
  • Early error detection
  • Easier refactoring
  • Improved code maintainability
  • Supports latest ECMAScript features
  • Excellent for large-scale applications

Installation

npm install -g typescript

Check version

tsc --version

Compile a file

tsc app.ts

Basic Types

number

const age: number = 25;

string

const name: string = "John";

boolean

const isAdmin: boolean = true;

null

const value: null = null;

undefined

const value: undefined = undefined;

bigint

const amount: bigint = 100n;

symbol

const id: symbol = Symbol();

any

  • Disables type checking.
  • Avoid using unless absolutely necessary.

let value: any = 10;

value = "Hello";
value = true;

unknown

  • Safer alternative to any.
  • Must check the type before using.
let value: unknown = "Hello";

if (typeof value === "string") {
    console.log(value.toUpperCase());
}

void

  • Used when a function doesn't return anything.
function printName(name: string): void {
    console.log(name);
}

never

  • Represents values that never occur.
function throwError(message: string): never {
    throw new Error(message);
}

Type Inference

  • TypeScript automatically detects the variable type.
const name = "John";
const age = 25;
const active = true;

Arrays

const numbers: number[] = [1, 2, 3];

Tuple

  • Fixed length array with different types.

Union Type

Literal Type

type Direction = "left" | "right" | "top" | "bottom";

Type Assertion

const text = value as string;

Functions

Function with Types

function add(a: number, b: number): number {
    return a + b;
}

Optional Parameter

function greet(name?: string) {
    console.log(name);
}

Default Parameter

function greet(name = "Guest") {
    console.log(name);
}

Rest Parameter

function sum(...numbers: number[]) {
    return numbers.reduce((a, b) => a + b, 0);
}

Arrow Function

const multiply = (a: number, b: number): number => {
    return a * b;
};

Interface

  • Defines the structure of an object.

    id: number;
    name: string;
}

const employee: Employee = {
    id: 1,
    name: "John"

Optional Property

    name: string;
    age?: number;

Readonly Property

    readonly id: number;
    name: string;

Type Alias

Classes

Basic Class

class Person {

    name: string;

    constructor(name: string) {
        this.name = name;
    }

}

Access Modifiers

class Employee {

    public name: string;

    private salary: number;

    protected department: string;

    constructor(name: string) {
        this.name = name;
    }

}

Inheritance

class Animal {

    move() {}

}

class Dog extends Animal {

    bark() {}

}

Abstract Class

abstract class Shape {

    abstract area(): number;

}

Implement Interface

interface Animal {

    sound(): void;

}

class Dog implements Animal {

    sound() {
        console.log("Bark");
    }

}

Generics

Generic Function

function identity<T>(value: T): T {
    return value;
}

Generic Interface

interface ApiResponse<T> {

    data: T;

    success: boolean;

}

Generic Class

class Box<T> {

    constructor(public value: T) {}

}

Utility Operators

keyof

interface User {

    name: string;

    age: number;

}

type UserKeys = keyof User;

typeof

const user = {
    name: "John"
};

type User = typeof user;

Readonly

Pick

type Scores = Record<string, number>;

Modern Features

Optional Chaining

const city = user?.address?.city;

Nullish Coalescing

const username = name ?? "Guest";

Type Narrowing

  • TypeScript narrows the type after checking.
function print(value: string | number) {

    if (typeof value === "string") {
        console.log(value.toUpperCase());
    } else {
        console.log(value.toFixed(2));
    }

}

    "compilerOptions": {

        "module": "ESNext",
        "strict": true,
        "rootDir": "./src",

    }
}
